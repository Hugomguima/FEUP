// Algoritmo ganancioso
int function() {
    int lastDist = pos[0];
    int valorT += valor[0];
    for(int i = 0; i < pos.size(); i++) {
        if ((lastDist - pos[i]) >= 5) {
            lastDist = pos[i];
            valorT += valor[i];
        }
    }
}
